addi x2, x0, -1888   /* Initiate LSU address for LCD */

lui  x3,      524288 /* Turn on the LCD */
addi x3, x3,  1080   /* 8-bit, 2-line mode, E = 1 */
sw   x3,      0(x2)  /* Send to LCD */
jal  x1,      delay  /* Delay a few miliseconds */

lui  x3,      524288
addi x3, x3,  56     /* 8-bit, 2-line mode, E = 0 */
sw   x3,      0(x2)
jal  x1,      delay

lui  x3,      524288
addi x3, x3,  1036   /* Cursor off, E = 1 */
sw   x3,      0(x2)
jal  x1,      delay

lui  x3,      524288
addi x3, x3,  12     /* Cursor off, E = 0 */
sw   x3,      0(x2)
jal  x1,      delay

addi x5, x0,  16

delay:
lui  x4,      17
addi x4, x4,  367
d:

bne  x4, x0,  d
jalr x0,      0(x1)
